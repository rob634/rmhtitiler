# =============================================================================
# geotiler — Production Dependencies (titiler-pgstac 2.1.0)
# =============================================================================
# Installed ON TOP of the base image (ghcr.io/stac-utils/titiler-pgstac:2.1.0)
# which already provides: titiler-pgstac, titiler-core, rio-tiler, rasterio,
# fastapi, starlette, uvicorn, psycopg, pydantic, pydantic-settings, etc.
#
# NOTE: asyncpg is NOT in the base image — we install it explicitly.
#
# The Dockerfile uses: pip install -r requirements.txt

# --- Azure Authentication -------------------------------------------------
# OAuth tokens via Managed Identity for storage and PostgreSQL
azure-identity>=1.16.1
# Key Vault access for password retrieval (key_vault auth mode)
azure-keyvault-secrets>=4.8.0

# --- Xarray / Zarr --------------------------------------------------------
# Multidimensional data support (Zarr v3 only — ETL converts upstream formats)
# titiler.xarray 1.2.x uses obstore (Rust object_store) for Azure blob access.
# Capped at <2.0 because titiler.xarray 2.0.0 requires titiler-core 2.0 +
# rio-tiler 9.x, which conflicts with the base image (titiler-core 1.2.0,
# rio-tiler 8.0.5).
titiler.xarray>=1.2.0,<2.0
# xarray must be >=2024.10.0 for Zarr v3 format support (dimension_names).
# The base image ships an older xarray that only reads Zarr v2 (_ARRAY_DIMENSIONS).
xarray>=2024.10.0

# --- Monitoring & Observability --------------------------------------------
# Hardware metrics in /health endpoint
psutil>=5.9.0
# Application Insights integration (set APPLICATIONINSIGHTS_CONNECTION_STRING)
azure-monitor-opentelemetry>=1.6.0

# --- Async PostgreSQL (not in base image) ----------------------------------
# Required by tipg and stac-fastapi-pgstac for async connection pools
asyncpg>=0.29.0

# --- OGC Features + Vector Tiles (TiPG) -----------------------------------
# OGC API - Features and OGC API - Tiles for PostGIS tables
# https://github.com/developmentseed/tipg
tipg>=1.3.1

# --- STAC API --------------------------------------------------------------
# Catalog browsing and search via stac-fastapi-pgstac
stac-fastapi.pgstac>=5.0.0,<7.0

# --- H3 Explorer (optional, gated by ENABLE_H3_DUCKDB) --------------------
duckdb>=1.1.0
